po zacati github codespace
# 1. Prejdi do projektu
cd /workspaces/PatentExpirationAPI

# 2. Naštartuj databázu (v jednom termináli)
docker-compose up db

- necham bezat terminal -nastartovala sa databaza
- spustim novy terminal
- pokracujem bodom 3

# 3. Aktivuj virtual environment
source venv/bin/activate

# 4. Spusti API
uvicorn app.main:app --reload

-ak chcem restarovat API -
    # Zastav uvicorn (Ctrl+C)
    # Potom znova spusti:
    uvicorn app.main:app --reload


    Python používa .pyc cache súbory!
Vyčisti ich:
bash# Zastav API (Ctrl+C)

# Vymaž všetky .pyc súbory
find . -type d -name __pycache__ -exec rm -rf {} + 2>/dev/null
find . -name "*.pyc" -delete

# Reštartuj API
uvicorn app.main:app --reload

overenie v novom terminali
curl "http://localhost:8000/api/v1/status?patent=EP0683520"
